<script>
    // Event Listeners untuk update real-time
    const inputs = document.querySelectorAll('input, textarea');
    inputs.forEach(input => {
        input.addEventListener('input', calculateSEO);
    });

    function copyToClipboard(elementId) {
        const text = document.getElementById(elementId).innerText;
        navigator.clipboard.writeText(text).then(() => {
            alert('Teks berhasil disalin!');
        });
    }

    function calculateSEO() {
        const keyword = document.getElementById('mainKeyword').value.trim();
        const title = document.getElementById('videoTitle').value.trim();
        const desc = document.getElementById('videoDesc').value.trim();

        if (!keyword) return;

        // 1. Generate Optimized Title
        let optimizedT = "";
        if (title.toLowerCase().includes(keyword.toLowerCase())) {
            optimizedT = `✅ Judul Anda sudah mengandung Keyword! (Coba tambahkan emoji atau angka, misal: [2024])`;
        } else {
            optimizedT = `💡 Saran Judul: ${keyword} - Cara Mudah & Cepat | Tutorial Lengkap 2024`;
        }
        document.getElementById('optimizedTitle').innerText = optimizedT;

        // 2. Generate Optimized Description
        // Format: Hook (Keyword), Isi, CTA, Tags
        const socialLinks = `\n\n🔗 Follow Saya:\nInstagram: https://instagram.com/\nTikTok: https://tiktok.com/@\n\n#SEO #YouTubeIndonesia #${keyword.replace(/\s/g, '')}`;
        document.getElementById('optimizedDesc').innerText = `${keyword} - Di video ini kita akan membahas tentang ${keyword}. Pastikan tonton sampai habis!\n\n👇 Ringkasan:\n- Apa itu ${keyword}?\n- Tips terbaik ${keyword}\n\n${socialLinks}`;

        // 3. Generate Tags
        const tags = [
            keyword, 
            `${keyword} tutorial`, 
            `${keyword} indonesia`, 
            `cara ${keyword}`, 
            `tips ${keyword}`,
            `review ${keyword}`,
            `${keyword} 2024`,
            `${keyword} mudah`
        ].join(', ');
        document.getElementById('tagsResult').innerText = tags;

        // 4. Calculate SEO Score
        let score = 0;
        
        // Cek Keyword di Judul (Bobot 40)
        if (title.toLowerCase().includes(keyword.toLowerCase())) score += 40;
        
        // Cek Panjang Judul (Ideal 60 char) (Bobot 20)
        if (title.length >= 40 && title.length <= 70) score += 20;
        else if (title.length > 0) score += 10;

        // Cek Keyword di Deskripsi (Bobet 20)
        if (desc.toLowerCase().includes(keyword.toLowerCase())) score += 20;

        // Cek Panjang Deskripsi (Bobet 20)
        if (desc.length > 200) score += 20;
        else if (desc.length > 50) score += 10;

        // Update UI Score
        const scoreEl = document.getElementById('seoScore');
        scoreEl.innerText = score + "/100";
        
        if(score >= 80) scoreEl.style.color = "green";
        else if(score >= 50) scoreEl.style.color = "orange";
        else scoreEl.style.color = "red";
    }
</script>
